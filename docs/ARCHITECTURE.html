📄 Page Class Example
export class LoginPage { constructor(private page: Page) {} // Locators as arrow functions usernameInput = () => this.page.locator('#username'); passwordInput = () => this.page.locator('#password'); submitBtn = () => this.page.getByRole('button'); // Action methods async fillUsername(user: string) { await this.usernameInput().fill(user); } }
⚙️ Module Class Example
export class LoginModule { constructor( private page: Page, private loginPage: LoginPage ) {} // Business logic orchestration async doLogin() { await this.loginPage.fillUsername('user'); await this.loginPage.fillPassword('pass'); await this.loginPage.clickSubmit(); } }
✏️ Test Spec Example
import { test } from '@fixtures'; test.describe('@P0 Login', () => { test('should login successfully', async ({ loginModule }) => { await test.step('Login', async () => { await loginModule.doLogin(); }); }); });
⚠️ LAYER 7: Common Mistakes & Code Review
🚫 Common Mistakes to Avoid
❌ Locators in Modules
// WRONG - Module
this.page.locator('#btn')
// RIGHT - Use Page
this.loginPage.submitBtn()
❌ Logic in Pages
// WRONG - Page
if (condition)
  doSomething()
// RIGHT - Module
handles logic
module.handleCondition()
❌ Missing test.step()
// WRONG
await login();
await navigate();
// RIGHT
await test.step('Login',
  ...)
✅ Code Review Checklist

📄 Page Class:

  • Locators as arrow functions
  • No business logic
  • Private page property
  • Named exports only
  • Single responsibility

⚙️ Module Class:

  • Uses Page class methods
  • No direct locators
  • Business logic only
  • Async/await pattern
  • Error handling

✏️ Test Spec:

  • Has tags (@P0, @Login)
  • Uses test.step()
  • Has beforeEach/afterEach
  • Screenshots attached
  • Proper cleanup

📋 General:

  • Correct folder location
  • Uses path aliases
  • No hardcoded values
  • JSDoc comments
  • No console.log
